import os
import requests
import speech_recognition as sr
from groq import Groq
from dotenv import load_dotenv

load_dotenv()

# --- API Keys ---
GROQ_API_KEY = os.getenv("GROQ_API_KEY")
ELEVENLABS_API_KEY = os.getenv("ELEVENLABS_API_KEY")

# --- ElevenLabs Voice (Jessica) ---
VOICE_ID = "cgSgspJ2msm6clMCkdW9"

# --- Friday's Brain ---
groq_client = Groq(api_key=GROQ_API_KEY)

# --- Conversation memory (keeps context across turns) ---
conversation_history = [
    {
        "role": "system",
        "content": """You are Friday, an advanced AI assistant — confident, intelligent, and occasionally witty.
You always address the user as 'Boss'.
Keep responses concise and conversational — 1 to 3 sentences max unless asked for detail.
You speak like the AI assistant from Iron Man. You are helpful, sharp, and loyal."""
    }
]

# ─────────────────────────────────────────
def speak(text):
    """Convert text to speech using ElevenLabs and play it."""
    print(f"🔊 Friday: {text}")
    
    url = f"https://api.elevenlabs.io/v1/text-to-speech/{VOICE_ID}"
    headers = {
        "xi-api-key": ELEVENLABS_API_KEY,
        "Content-Type": "application/json"
    }
    payload = {
        "text": text,
        "model_id": "eleven_turbo_v2_5",
        "voice_settings": {
            "stability": 0.5,
            "similarity_boost": 0.75
        }
    }
    
    response = requests.post(url, json=payload, headers=headers)
    
    if response.status_code == 200:
        with open("friday_speech.mp3", "wb") as f:
            f.write(response.content)
        os.system("start friday_speech.mp3")
    else:
        print(f"❌ Voice error: {response.status_code} {response.text}")

# ─────────────────────────────────────────
def listen():
    """Listen to microphone and return transcribed text."""
    recognizer = sr.Recognizer()
    
    with sr.Microphone() as source:
        recognizer.adjust_for_ambient_noise(source, duration=0.5)
        print("🎙️ Listening...")
        
        try:
            audio = recognizer.listen(source, timeout=8, phrase_time_limit=12)
            text = recognizer.recognize_google(audio)
            print(f"👤 Boss: {text}")
            return text
        except sr.WaitTimeoutError:
            return None
        except sr.UnknownValueError:
            return None
        except sr.RequestError as e:
            print(f"❌ Mic error: {e}")
            return None

# ─────────────────────────────────────────
def think(user_input):
    """Send user input to Groq and get Friday's response."""
    conversation_history.append({
        "role": "user",
        "content": user_input
    })
    
    response = groq_client.chat.completions.create(
        model="llama-3.3-70b-versatile",
        messages=conversation_history,
        max_tokens=200
    )
    
    reply = response.choices[0].message.content
    
    conversation_history.append({
        "role": "assistant",
        "content": reply
    })
    
    return reply

# ─────────────────────────────────────────
def main():
    print("=" * 50)
    print("       F.R.I.D.A.Y  —  Online")
    print("=" * 50)
    print("Say 'goodbye' or 'exit' to shut down.")
    print()
    
    # Friday introduces herself
    speak("Good day, Boss. Friday is online and fully operational. How can I assist you?")
    
    while True:
        user_input = listen()
        
        if not user_input:
            continue
        
        # Check for exit commands
        if any(word in user_input.lower() for word in ["goodbye", "exit", "shut down", "goodbye friday"]):
            speak("Understood, Boss. Shutting down. Stay brilliant.")
            break
        
        # Get Friday's response and speak it
        response = think(user_input)
        speak(response)

if __name__ == "__main__":
    main()
